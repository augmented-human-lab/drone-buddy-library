 44    def get_feature(self, image) -> list:
 45        """
 46        Detect hands in an image.
 47
 48        Args:
 49            image (list): The frame to detect the hand in.
 50
 51        Returns:
 52            list | bool: Return the list of the landmark of one hand in the frame.
 53                         Return False if no hand is detected.
 54        """
 55        copied_image = image.copy()
 56        hand_landmarks = []
 57        with mp.solutions.hands.Hands(
 58                static_image_mode=True,
 59                max_num_hands=1,
 60                min_detection_confidence=0.5) as hands:
 61            results = hands.process(cv2.cvtColor(copied_image, cv2.COLOR_BGR2RGB))
 62            if results.multi_hand_landmarks:
 63                for hand_landmarks_ in results.multi_hand_landmarks:
 64                    for landmark in hand_landmarks_.landmark:
 65                        hand_landmarks.append([landmark.x, landmark.y])
 66            else:
 67                hand_landmarks = False
 68
 69        self.hand_landmark = hand_landmarks
 70        return hand_landmarks

 72    def count_fingers(self, frame, show_feedback=False) -> int:
 73        """
 74        Count the number of fingers in a frame.
 75
 76        Args:
 77            frame (np.array): The frame to count fingers in.
 78            show_feedback (bool): Whether to show the processed frame.
 79
 80        Returns:
 81            int: The number of fingers in the frame.
 82        """
 83        global p_time
 84
 85        c_time = time.time()
 86        fps = 1 / (c_time - p_time)
 87        p_time = c_time
 88
 89        cv2.putText(frame, f'FPS:{int(fps)}', (400, 70), cv2.FONT_HERSHEY_PLAIN, 3, (0, 0, 255), 3)
 90
 91        img_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
 92        results = hands.process(frame)
 93
 94        finger_count = 0
 95
 96        if results.multi_hand_landmarks:
 97            for hand_landmarks in results.multi_hand_landmarks:
 98                for id, lm in enumerate(hand_landmarks.landmark):
 99                    h, w, c = frame.shape
100                    cx, cy = int(lm.x * w), int(lm.y * h)
101                    if id == 12:
102                        cv2.circle(frame, (cx, cy), 10, (255, 255, 0), cv2.FILLED)
103                        cv2.putText(frame, f'id12z:{int(lm.z * (-1000))}', (50, 250), cv2.FONT_HERSHEY_PLAIN, 3,
104                                    (255, 0, 0),
105                                    3)
106                    mpDraw.draw_landmarks(frame, hand_landmarks, mpHands.HAND_CONNECTIONS)
107                hand_index = results.multi_hand_landmarks.index(hand_landmarks)
108                hand_label = results.multi_handedness[hand_index].classification[0].label
109                hand_landmarks = []
110                for landmarks in hand_landmarks.landmark:
111                    hand_landmarks.append([landmarks.x, landmarks.y])
112                if hand_label == "Left" and hand_landmarks[4][0] > hand_landmarks[3][0]:
113                    finger_count = finger_count + 1
114                elif hand_label == "Right" and hand_landmarks[4][0] < hand_landmarks[3][0]:
115                    finger_count = finger_count + 1
116                if hand_landmarks[8][1] < hand_landmarks[6][1]:
117                    finger_count = finger_count + 1
118                if hand_landmarks[12][1] < hand_landmarks[10][1]:
119                    finger_count = finger_count + 1
120                if hand_landmarks[16][1] < hand_landmarks[14][1]:
121                    finger_count = finger_count + 1
122                if hand_landmarks[20][1] < hand_landmarks[18][1]:
123                    finger_count = finger_count + 1
124        cv2.rectangle(frame, (390, 10), (470, 120), (0, 255, 255), cv2.FILLED)
125        cv2.putText(frame, str(finger_count), (400, 100), cv2.FONT_HERSHEY_SIMPLEX, 3, (255, 0, 0), 10)
126
127        if show_feedback:
128            cv2.imshow("Image", frame)
129
130        return finger_count
